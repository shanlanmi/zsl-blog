----
title: 用正则转化成千分位数字
date: 2016-12-04 18:06:54
tags:
- JavaScirpt
----
```js
let milliFormat = (() => {
    const DIGIT_PATTERN = /(^|\s)\d+(?=\.?\d*($|\s))/g
    const MILLI_PATTERN = /(?=(?!\b)(\d{3})+$)/g

    return (input) => input && input.toString()
        .replace(DIGIT_PATTERN, (m) => m.replace(MILLI_PATTERN, ','))
})()
```
